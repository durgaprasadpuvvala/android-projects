1    package com.example.busapp;
2    
3    import android.os.Bundle;
4    import androidx.fragment.app.FragmentActivity;
5    import com.google.android.gms.maps.*;
6    import com.google.android.gms.maps.model.*;
7    import com.google.firebase.database.*;
8    
9    public class StudentDashboardActivity extends FragmentActivity
10           implements OnMapReadyCallback {
11   
12       GoogleMap map;
13   
14       @Override
15       protected void onCreate(Bundle b) {
16           super.onCreate(b);
17           setContentView(R.layout.activity_student_dashboard);
18   
19           SupportMapFragment frag =
20                   (SupportMapFragment) getSupportFragmentManager()
21                           .findFragmentById(R.id.map);
22   
23           frag.getMapAsync(this);
24       }
25   
26       @Override
27       public void onMapReady(GoogleMap googleMap) {
28           map = googleMap;
29   
30           FirebaseDatabase.getInstance()
31                   .getReference("busLocation/BUS_101")
32                   .addValueEventListener(new ValueEventListener() {
33                       @Override
34                       public void onDataChange(DataSnapshot s) {
35                           if (!s.exists()) return;
36   
37                           double lat = s.child("latitude").getValue(Double.class);
38                           double lng = s.child("longitude").getValue(Double.class);
39   
40                           LatLng bus = new LatLng(lat, lng);
41                           map.clear();
42                           map.addMarker(new MarkerOptions().position(bus).title("Bus Location"));
43                           map.moveCamera(CameraUpdateFactory.newLatLngZoom(bus, 15));
44                       }
45   
46                       @Override public void onCancelled(DatabaseError e) {}
47                   });
48       }
49   }

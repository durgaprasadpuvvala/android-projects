1    package com.example.busapp;
2    
3    import android.content.Intent;
4    import android.os.Bundle;
5    import android.widget.*;
6    import androidx.appcompat.app.AppCompatActivity;

8    
9    public class StudentLoginActivity extends AppCompatActivity {
10   
11       EditText email, password;
12       Button login;
13       TextView register;
14       FirebaseAuth auth;
15   
16       @Override
17       protected void onCreate(Bundle b) {
18           super.onCreate(b);
19           setContentView(R.layout.activity_student_login);
20   
21           email = findViewById(R.id.etEmail);

24           register = findViewById(R.id.tvRegister);
25   
26           auth = FirebaseAuth.getInstance();
27   
28           login.setOnClickListener(v ->
29                   auth.signInWithEmailAndPassword(
30                           email.getText().toString(),
31                           password.getText().toString()
32                   ).addOnSuccessListener(res -> {
33                       startActivity(new Intent(this, StudentDashboardActivity.class));
34                       finish();
35                   })
36           );
37   
38           register.setOnClickListener(v ->
39                   startActivity(new Intent(this, StudentRegistrationActivity.class)));
40       }

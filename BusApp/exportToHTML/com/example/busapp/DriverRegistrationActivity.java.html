1    package com.example.busapp;
2    
3    import android.content.Intent;
4    import android.os.Bundle;
5    import android.widget.*;
6    import androidx.appcompat.app.AppCompatActivity;
7    
8    import com.google.firebase.auth.FirebaseAuth;
9    import com.google.firebase.firestore.FirebaseFirestore;
10   
11   import java.util.HashMap;
12   import java.util.Map;
13   
14   public class DriverRegistrationActivity extends AppCompatActivity {
15   
16       EditText name, email, password, bus;
17       Button register;
18       FirebaseAuth auth;
19       FirebaseFirestore db;
20   
21       @Override
22       protected void onCreate(Bundle b) {
23           super.onCreate(b);
24           setContentView(R.layout.activity_driver_registration);
25   
26           name = findViewById(R.id.name);
27           email = findViewById(R.id.email);
28           password = findViewById(R.id.password);
29           bus = findViewById(R.id.bus);
30           register = findViewById(R.id.register);
31   
32           auth = FirebaseAuth.getInstance();
33           db = FirebaseFirestore.getInstance();
34   
35           register.setOnClickListener(v -> {
36   
37               String sName = name.getText().toString().trim();
38               String sEmail = email.getText().toString().trim();
39               String sPassword = password.getText().toString().trim();
40               String sBus = bus.getText().toString().trim();
41   
42               if (sName.isEmpty() || sEmail.isEmpty() || sPassword.isEmpty() || sBus.isEmpty()) {
43                   Toast.makeText(this, "All fields required", Toast.LENGTH_SHORT).show();
44                   return;
45               }
46   
47               auth.createUserWithEmailAndPassword(sEmail, sPassword)
48                       .addOnSuccessListener(result -> {
49   
50                           String uid = auth.getCurrentUser().getUid();
51   
52                           Map<String, Object> map = new HashMap<>();
53                           map.put("name", sName);
54                           map.put("email", sEmail);
55                           map.put("bus", sBus);
56                           map.put("role", "driver");
57   
58                           db.collection("drivers").document(uid)
59                                   .set(map)
60                                   .addOnSuccessListener(unused -> {
61                                       Toast.makeText(this, "Driver Registered", Toast.LENGTH_SHORT).show();
62                                       startActivity(new Intent(this, DriverLoginActivity.class));
63                                       finish();
64                                   });
65                       })
66                       .addOnFailureListener(e ->
67                               Toast.makeText(this, e.getMessage(), Toast.LENGTH_LONG).show()
68                       );
69           });
70       }
71   }
